SubProceso  masaCorporal 
    Definir peso,estatura,imc Como Real;
        
    Escribir "Digite el número de su peso en Kg: ";
    Leer peso;
    escribir "Digite el número de su estatura en metros:";
    Leer estatura;
    
    imc ← peso / (estatura  ↑ 2);
    Escribir "Su imc (Indice de Masa Corporal) es de : ", imc;
    
    si imc < 18.5 Entonces
        Escribir "La persona se encuentra bajo de peso";
    FinSi
    
    Si imc  ≥ 18.5 ∧ imc < 25 Entonces
        Escribir "La persona se encuentra en su peso normal";
    FinSi
    
    si imc ≥ 25 ∧ imc < 30 Entonces
        Escribir "La persona se encuentra con sobrepeso";
    FinSi
    
    Si imc > 30 Entonces
        Escribir "La persona se encuntra con obesidad";
    FinSi    
    
FinSubProceso
